## Variable
``` golang
package main

import "fmt"

func main() {
	var day int = 15
	var month = "October"
	year := 2021

	fmt.Println(day)
	fmt.Println(month)
	fmt.Println(year)
}
```
```
$ go run .
> 15
> October
> 2021
```

## Update Variable
``` golang
package main

import "fmt"

func main() {
	var day int = 15
	var month = "October"
	year := 2021

	fmt.Println(day)
	fmt.Println(month)
	fmt.Println(year)

	// Update
	day = 20
	month = "November"
	year = 2025
	fmt.Println("------------------------------------------")
	fmt.Println(day)
	fmt.Println(month)
	fmt.Println(year)
}
```
```
$ go run .
> 15
> October
> 2021
> ------------------------------------------
> 20
> November
> 2025
```

## Deklarasi Multiple Variable
``` golang
func main() {
	var (
		firstName = "Yosho"
		lastName  = "Prasetiyo"
	)

	fmt.Println(firstName)
	fmt.Println(lastName)
}
```

```
$ go run main.go
> Yosho
> Prasetiyo
```
