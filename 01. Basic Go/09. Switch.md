## Switch
``` golang
package main

import "fmt"

func main() {
	money := 50000
	switch money {
	case 0:
		fmt.Println("Belum beruntung")

	case 1000, 2000, 5000:
		fmt.Println("Lumayan")

	case 10000, 20000:
		fmt.Println("Beuntung")

	case 50000, 100000:
		fmt.Println("Sangat beruntung")

	default:
		fmt.Println("Mungkin Anda tidak sedang mengikuti undian")
	}
}
```
```
$ go run .
> Sangat beruntung
```
