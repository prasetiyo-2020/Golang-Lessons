## Integer and Mathematical operations
``` golang
package main

import "fmt"

func main() {
	var score int8 = 50

	fmt.Println(score + 1)
	fmt.Println(score - 1)
	fmt.Println(score * 2)
	fmt.Println(score / 2)
	fmt.Println(score % 2)
}

```
```
$ go run .
$ go run main.go
> 51
> 49
> 100
> 25
> 0
```

## Augmented Assigments
| Mathematical operations  | Augmented Assigments |
| :-------------: | :-------------: |
| a = a + 10  | a += 10  |
| a = a - 10  | a -= 10  |
| a = a * 10  | a *= 10  |
| a = a / 10  | a /= 10  |
| a = a % 10  | a %= 10  |

``` golang
package main

import "fmt"

func main() {
	var score1 int8 = 10
	score1 += 2

	var score2 int8 = 10
	score2 -= 2

	var score3 int8 = 10
	score3 *= 2

	var score4 int8 = 10
	score4 /= 2

	var score5 int8 = 10
	score5 %= 2

	fmt.Println(score1)
	fmt.Println(score2)
	fmt.Println(score3)
	fmt.Println(score4)
	fmt.Println(score5)
}

```
```
$ go run main.go
> 12
> 8
> 20
> 5
> 0
```

## Unary Operator
| Operator  | Keterangan |
| :-------------: | :-------------: |
| ++  | a = a + 1  |
| --  | a = a - 1  |
| -  | Negative  |
| + | Positive  |
| !  | Boolean kebalikan  |

``` golang
package main

import "fmt"

func main() {
	var score1 int8 = 10
	score1++
	fmt.Println(score1)

	var score2 int8 = 10
	score2--
	fmt.Println(score2)

	var score3 int8 = -10
	fmt.Println(score3)

	var score4 int8 = 10
	fmt.Println(score4)
}
```
```
$ go run main.go
> 11
> 9  
> -10
> 10
```
