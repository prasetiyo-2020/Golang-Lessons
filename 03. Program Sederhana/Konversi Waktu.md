## Konversi Waktu

``` golang
// Konversi Menit ke Detik
func MenitKeDetik(menit int) int {
	return menit * 60
}

// Konversi Detik ke Menit
func DetikKeMenit(detik int) int {
	return detik / 60
}

// Konversi Jam ke Menit
func JamKeMenit(jam int) int {
	return jam * 60
}

// Konversi Menit ke Jam
func MenitKeJam(menit int) int {
	return menit / 60
}

}

}
```
